// 各センサへ送る模擬データ用の変数
uint16_t simulatedLidarRange = 1200;  // 1200mm (1.2m)
uint16_t simulatedUltSoundDist = 85;  // 85cm

void initCAN() {
    // TWAI（CAN）コントローラの初期化設定
    twai_general_config_t g_config = TWAI_GENERAL_CONFIG_DEFAULT(CAN_TX_PIN, CAN_RX_PIN, TWAI_MODE_NORMAL);
    
    // 通信速度の設定 (ここではロボティクス等で一般的な 500kbps に設定)
    twai_timing_config_t t_config = TWAI_TIMING_CONFIG_500KBITS();
    
    // フィルター設定 (すべてのIDを受け入れる)
    twai_filter_config_t f_config = TWAI_FILTER_CONFIG_ACCEPT_ALL();

    // ドライバーのインストール
    if (twai_driver_install(&g_config, &t_config, &f_config) == ESP_OK) {
        Serial.println("[OK] TWAI (CAN) Driver Installed");
    } else {
        Serial.println("[ERR] Failed to install TWAI Driver");
        return;
    }

    // CAN通信の開始
    if (twai_start() == ESP_OK) {
        Serial.println("[OK] TWAI (CAN) Driver Started");
    } else {
        Serial.println("[ERR] Failed to start TWAI Driver");
    }
}

void sendSensorData(uint32_t id, uint8_t* data, uint8_t dlc) {
    twai_message_t message;
    message.identifier = id;           // 送信先CAN ID
    message.extd = 0;                  // 0: 標準ID(11bit), 1: 拡張ID(29bit)
    message.rtr = 0;                   // 0: データフレーム, 1: リモートフレーム
    message.data_length_code = dlc;    // データ長 (0〜8バイト)
    
    // データのコピー
    for (int i = 0; i < dlc; i++) {
        message.data[i] = data[i];
    }

    // 待ち時間なし（明示的なノンブロッキング）で送信キューに入れる
    esp_err_t res = twai_transmit(&message, pdMS_TO_TICKS(0));
    if (res == ESP_OK) {
        Serial.printf("[TX] Sent to ID 0x%02X, DLC: %d\n", id, dlc);
    } else {
        Serial.printf("[TX ERR] Failed to send to ID 0x%02X (Error: %d)\n", id, res);
    }
}

void setup() {
    Serial.begin(115200);
    delay(2000); // シリアル表示が間に合うように少し待つ
    Serial.println("--- ESP32 TWAI (CAN) TX Test ---");

    initCAN();
}

void loop() {
    // ---- 1. LiDAR 宛てのデータ作成と送信 ----
    // 例として2バイト（16bit）の距離データを 上位・下位バイトに分解して送る
    uint8_t lidarData[2];
    lidarData[0] = (simulatedLidarRange >> 8) & 0xFF; // 上位バイト
    lidarData[1] = simulatedLidarRange & 0xFF;        // 下位バイト
    
    sendSensorData(lidarCANaddr, lidarData, 2);

    // ---- 2. 超音波センサ 宛てのデータ作成と送信 ----
    // 例として1バイトの距離データと、適当なステータス（例: 0xAA）を送る
    uint8_t ultSoundData[2];
    ultSoundData[0] = simulatedUltSoundDist & 0xFF;
    ultSoundData[1] = 0xAA; // 正常ステータスを示すダミー値
    
    sendSensorData(ultSoundCANaddr, ultSoundData, 2);

    // テスト用に送信する数値を少しずつ変化させる
    simulatedLidarRange += 5;
    if (simulatedLidarRange > 5000) simulatedLidarRange = 1000;

    simulatedUltSoundDist += 1;
    if (simulatedUltSoundDist > 400) simulatedUltSoundDist = 20;

    // 500ミリ秒（0.5秒）ごとに繰り返し送信
    delay(500);
}
